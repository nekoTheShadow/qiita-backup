    素因数分解をMySQLで実装した

    仕事の会議中に暇つぶしに作成したので、ここにさらしておきます。実用性はありません（´・ω・｀）

    100以下の自然数について、素因数分解するSQLです。MySQL8にて稼働確認をしています。

    WITH RECURSIVE 
numbers(x) AS (
    SELECT 2
  UNION ALL
    SELECT x+1 FROM numbers WHERE x < 100
),
factors(x, y, i, divided) AS (
    SELECT x, x, 2, 0 FROM numbers
  UNION ALL
    SELECT
      x,
      CASE WHEN y MOD i = 0 THEN y DIV i ELSE y   END, 
      CASE WHEN y MOD i = 0 THEN i       ELSE i+1 END,
      CASE WHEN y MOD i = 0 THEN 1       ELSE 0   END
    FROM factors
    WHERE y > 1
)
SELECT x, GROUP_CONCAT(i ORDER BY i SEPARATOR ' ') AS primes
FROM factors 
WHERE divided = 1
GROUP BY x
ORDER BY x;
